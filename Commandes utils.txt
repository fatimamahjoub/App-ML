### Installation (local)

# (optionnel) créer un environnement virtuel
python -m venv .venv

# macOS / Linux
source .venv/bin/activate

# Windows (PowerShell)
# .venv\Scripts\Activate.ps1

pip install -r requirements.txt


### Lancer l'application Streamlit

streamlit run app.py


### Données utilisées

# Par défaut l'app cherche le fichier Excel ici (dans le dossier de l'app) :
# streamlit-app-v2/data/Risque_data.xlsx
#
# Vous pouvez aussi uploader un fichier .xlsx ou .csv depuis l'UI.
